        tamanho_lista = int(input())
contador = 1
lista_pares = []
lista_impares = []
while(contador <= tamanho_lista):
    valores = int(input())
    if valores % 2 == 0:
        lista_pares.append(valores)
    else:
        lista_impares.append(valores)
    contador = contador + 1
lista_pares.sort()
lista_impares.sort(reverse=True)
for j in lista_pares:
    print(j)
for k in lista_impares:
    print(k)
