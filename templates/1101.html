Beecrowd 1101 Solution

        
while(True):
    soma = 0
    try:
      m = input().split()
      n = list(map(int, m))
      check = any(x <= 0 for x in n)
    except ValueError:
      continue
    if check == True:
        break
    else:
      for j in range(min(n), max(n) + 1):
        soma += j
        print(j, end=" ")
      print(f"Sum={soma}")
