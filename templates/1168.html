while(ativaLaco):
    try:
        casoTeste = int(input())
    except ValueError:
        continue
    for i in range(1, casoTeste + 1):
        soma = 0
        n = input().replace("", " ").strip().split()
        for j in n:
            if j in led.keys():
                soma += led.get(j)
        print(f"{soma} leds")
